(define (domain partial-observable-exploration)
  (:requirements :strips :typing)
  (:types location direction)
  (:predicates
    (at ?loc - location)
    (door-between ?loc1 - location ?loc2 - location ?dir - direction)
    (door-closed ?loc1 - location ?loc2 - location ?dir - direction)
    (door-open ?loc1 - location ?loc2 - location ?dir - direction)
    (explored ?loc - location)
  )

  (:action open-door
      :parameters (?loc1 - location ?loc2 - location ?dir - direction)
      :precondition (and (at ?loc1) (door-between ?loc1 ?loc2 ?dir) (door-closed ?loc1 ?loc2 ?dir))
      :effect (and (door-open ?loc1 ?loc2 ?dir) (not (door-closed ?loc1 ?loc2 ?dir)))
  )

  (:action move
      :parameters (?from - location ?to - location ?dir - direction)
      :precondition (and (at ?from) (door-between ?from ?to ?dir) (door-open ?from ?to ?dir))
      :effect (and (not (at ?from)) (at ?to) (explored ?to))
  )
