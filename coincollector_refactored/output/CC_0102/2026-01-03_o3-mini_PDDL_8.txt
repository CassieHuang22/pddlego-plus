  (:types location direction)
  (:predicates
    (at ?l - location)
    (adjacent ?from - location ?to - location ?dir - direction)
    (closed ?from - location ?to - location ?dir - direction)
    (opposite ?d1 - direction ?d2 - direction)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (adjacent ?loc1 ?loc2 ?dir) (closed ?loc1 ?loc2 ?dir))
    :effect (not (closed ?loc1 ?loc2 ?dir))
  )

  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from)
                       (or (and (adjacent ?from ?to ?dir) (not (closed ?from ?to ?dir)))
                           (exists (?d - direction)

(define (domain exploration)
  (:requirements :typing :disjunctive-preconditions :existential-preconditions :conditional-effects)
  (:types location direction)
  (:predicates
    (at ?l - location)
    (adjacent ?from - location ?to - location ?dir - direction)
    (closed ?from - location ?to - location ?dir - direction)
    (opposite ?d1 - direction ?d2 - direction)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (adjacent ?loc1 ?loc2 ?dir) (closed ?loc1 ?loc2 ?dir))
    :effect (and (not (closed ?loc1 ?loc2 ?dir))
                 (forall (?d - direction)
                   (when (opposite ?dir ?d)

  )

    :precondition (and (at ?from) (adjacent ?from ?to ?dir) (not (closed ?from ?to ?dir)))

(define (domain exploration)
  (:requirements :typing :disjunctive-preconditions :existential-preconditions :conditional-effects)
  (:types location direction)
  (:predicates
    (at ?l - location)
    (adjacent ?from - location ?to - location ?dir - direction)
    (closed ?from - location ?to - location ?dir - direction)
    (opposite ?d1 - direction ?d2 - direction)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (adjacent ?loc1 ?loc2 ?dir) (closed ?loc1 ?loc2 ?dir))
    :effect (and (not (closed ?loc1 ?loc2 ?dir))
                 (forall (?d - direction)
                   (when (opposite ?dir ?d)
                     (not (closed ?loc2 ?loc1 ?d)))))
  )

  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (adjacent ?from ?to ?dir) (not (closed ?from ?to ?dir)))
    :effect (and (not (at ?from)) (at ?to))
  )
