  (:types location direction)
  (:predicates
    (at ?loc - location)
    (door_closed ?loc1 - location ?loc2 - location ?dir - direction)
    (door_opened ?loc1 - location ?loc2 - location ?dir - direction)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (door_closed ?loc1 ?loc2 ?dir)
    :effect (and
              (not (door_closed ?loc1 ?loc2 ?dir))
              (door_opened ?loc1 ?loc2 ?dir)
            )
  )

  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (door_opened ?from ?to ?dir))
    :effect (and (not (at ?from)) (at ?to))
  )
)
