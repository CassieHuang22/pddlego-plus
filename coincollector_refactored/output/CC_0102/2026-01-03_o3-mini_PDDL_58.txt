  (:requirements :strips :typing)
  (:types location direction material)
  (:predicates
    (at ?loc - location)
    (adjacent ?loc1 - location ?loc2 - location ?dir - direction)
    (passable ?loc1 - location ?loc2 - location ?dir - direction)
    (door-closed ?loc1 - location ?loc2 - location ?dir - direction)
    (door-open ?loc1 - location ?loc2 - location ?dir - direction)
    (observed-door ?loc - location ?dir - direction ?mat - material)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) (door-closed ?loc1 ?loc2 ?dir))
    :effect (and
              (door-open ?loc1 ?loc2 ?dir)
              (passable ?loc1 ?loc2 ?dir)
              (not (door-closed ?loc1 ?loc2 ?dir))
            )
  )

  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (passable ?from ?to ?dir))
    :effect (and
              (at ?to)
              (not (at ?from))
            )
  )
)
