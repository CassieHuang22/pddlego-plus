    (door-type-at ?from - location ?to - location ?dir - direction ?type - door-type)
    (door-open ?from - location ?to - location ?dir - direction)
    (visited ?loc - location)
  )

    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
