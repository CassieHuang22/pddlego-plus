(define (domain exploration)
  (:requirements :strips :typing)
  (:types location direction door)

  (:predicates
    (at ?l - location)
    (connected ?from - location ?to - location ?dir - direction)
    (door-between ?from - location ?to - location ?dir - direction ?d - door)
    (door-open ?d - door)
    (door-closed ?d - door)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction ?d - door)
    :precondition (and (door-between ?loc1 ?loc2 ?dir ?d) (door-closed ?d))
    :effect (and (not (door-closed ?d)) (door-open ?d))
