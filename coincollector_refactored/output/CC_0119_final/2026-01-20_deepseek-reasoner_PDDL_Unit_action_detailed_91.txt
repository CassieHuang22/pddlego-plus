    (at ?l - location)

    (closed ?d - direction ?l1 - location ?l2 - location)
    (open ?d - direction ?l1 - location ?l2 - location)
  )
  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
