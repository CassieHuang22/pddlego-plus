(define (domain exploration-domain)
  (:requirements :strips :typing)
  (:types location direction)
  (:predicates 
    (at ?l - location)
    (connected ?from ?to - location ?dir - direction)
    (has-door ?from ?to - location ?dir - direction)
    (door-open ?from ?to - location ?dir - direction)
    (visited ?l - location)
  )
  (:action open-door
    :parameters (?loc1 ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) 
                       (has-door ?loc1 ?loc2 ?dir)
                       (not (door-open ?loc1 ?loc2 ?dir)))
    :effect (and (door-open ?loc1 ?loc2 ?dir))
  )
  (:action move
    :parameters (?from ?to - location ?dir - direction)
    :precondition (and (at ?from) 
                       (connected ?from ?to ?dir)

                           (door-open ?from ?to ?dir)))
    :effect (and (not (at ?from)) 
                 (at ?to)

  )
