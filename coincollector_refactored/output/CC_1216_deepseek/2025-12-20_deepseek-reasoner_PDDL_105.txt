(define (domain exploration-domain)
  (:requirements :strips :typing)
  (:types location direction)
  (:predicates
    (at ?l - location)
    (adjacent ?l1 - location ?l2 - location ?dir - direction)
    (door-closed ?l1 - location ?l2 - location ?dir - direction)
    (visited ?l - location)
  )
  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (adjacent ?loc1 ?loc2 ?dir)
                       (door-closed ?loc1 ?loc2 ?dir))
    :effect (not (door-closed ?loc1 ?loc2 ?dir))
  )
  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from)
                       (adjacent ?from ?to ?dir)
                       (not (door-closed ?from ?to ?dir)))
    :effect (and (not (at ?from))
                 (at ?to)
