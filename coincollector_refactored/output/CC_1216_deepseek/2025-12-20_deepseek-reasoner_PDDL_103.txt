(define (domain exploration-domain)
    (:requirements :strips :typing)
    (:types location direction)
    (:predicates
        (at ?l - location)
        (visited ?l - location)
        (door ?from - location ?to - location ?d - direction)
        (closed ?from - location ?to - location ?d - direction)
        (open ?from - location ?to - location ?d - direction)
    )
    (:action open-door
        :parameters (?loc1 - location ?loc2 - location ?dir - direction)
        :precondition (and (at ?loc1) (door ?loc1 ?loc2 ?dir) (closed ?loc1 ?loc2 ?dir))
        :effect (and (not (closed ?loc1 ?loc2 ?dir)) (open ?loc1 ?loc2 ?dir))
    )
    (:action move

        :precondition (and (at ?from) (door ?from ?to ?dir) (open ?from ?to ?dir))
        :effect (and (not (at ?from)) (at ?to) (visited ?to))
    )
