(define (domain exploration)
  (:requirements :strips :typing)
  (:types location direction)
  (:predicates
    (at ?l - location)
    (visited ?l - location)
    (adjacent ?from - location ?to - location ?d - direction)
    (closed-door ?from - location ?to - location ?d - direction)
  )
  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (adjacent ?from ?to ?dir) (not (closed-door ?from ?to ?dir)))
    :effect (and (not (at ?from)) (at ?to) (visited ?to))
  )
  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) (adjacent ?loc1 ?loc2 ?dir) (closed-door ?loc1 ?loc2 ?dir))
    :effect (and (not (closed-door ?loc1 ?loc2 ?dir)))
  )
