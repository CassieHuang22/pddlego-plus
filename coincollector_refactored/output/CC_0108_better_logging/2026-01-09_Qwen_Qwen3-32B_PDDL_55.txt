    (at ?l - location)
    (direction ?from - location ?to - location ?dir - direction)
    (closed ?from - location ?dir - direction)
    (open ?from - location ?dir - direction)
  )
  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) (closed ?loc1 ?dir))
    :effect (and (not (closed ?loc1 ?dir)) (open ?loc1 ?dir))
  )
  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (direction ?from ?to ?dir) (open ?from ?dir))
    :effect (and (not (at ?from)) (at ?to))
  )
)
