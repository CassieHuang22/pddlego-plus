  (:types location direction doortype)
  (:predicates
    (at ?l - location)
    (adjacent ?l1 - location ?l2 - location ?dir - direction)
    (door ?l1 - location ?l2 - location ?dir - direction)
    (closed ?l1 - location ?l2 - location ?dir - direction)
    (open ?l1 - location ?l2 - location ?dir - direction)
    (door-type ?l1 - location ?l2 - location ?dt - doortype)
  )

  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) (door ?loc1 ?loc2 ?dir) (closed ?loc1 ?loc2 ?dir))
    :effect (and (not (closed ?loc1 ?loc2 ?dir)) (open ?loc1 ?loc2 ?dir))
  )

  (:action move
    :parameters (?from - location ?to - location ?dir - direction)
    :precondition (and (at ?from) (adjacent ?from ?to ?dir) (open ?from ?to ?dir))
    :effect (and (not (at ?from)) (at ?to))
  )
)
