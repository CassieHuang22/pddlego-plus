  (:types location direction)
  (:predicates 
    (at ?l - location)
    (adjacent ?l1 - location ?l2 - location ?d - direction)
    (has-door ?l1 - location ?l2 - location ?d - direction)
    (door-open ?l1 - location ?l2 - location ?d - direction)
  )
  (:action open-door
    :parameters (?loc1 - location ?loc2 - location ?dir - direction)
    :precondition (and (at ?loc1) 
                       (has-door ?loc1 ?loc2 ?dir)
                       (not (door-open ?loc1 ?loc2 ?dir)))
    :effect (door-open ?loc1 ?loc2 ?dir)
  )
  (:action move
    :parameters (?from - location ?to - location ?dir - direction)

                       (adjacent ?from ?to ?dir)
