  (:types agent object receptacle microwaveReceptacle fridgeReceptacle)
  (:predicates
    (agent-at ?r - receptacle)
    (holding ?o - object)
    (in ?o - object ?r - receptacle)
    (on ?o - object ?r - receptacle)
    (open ?r - receptacle)
    (closed ?r - receptacle)
    (switched-on ?o - object)
    (heated ?o - object)
    (clean ?o - object)
    (cooled ?o - object)
    (sliced ?o - object)
  )

  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (agent-at ?from)
    :effect (and (not (agent-at ?from)) (agent-at ?to))
  )

  (:action OpenObject
    :parameters (?r - receptacle)
    :precondition (and (agent-at ?r) (closed ?r))
    :effect (and (open ?r) (not (closed ?r)))
  )

  (:action CloseObject
    :parameters (?r - receptacle)
    :precondition (and (agent-at ?r) (open ?r))
    :effect (and (closed ?r) (not (open ?r)))
  )

  (:action PickupObject
    :parameters (?o - object ?r - receptacle)
    :precondition (and (agent-at ?r) (in ?o ?r))
    :effect (and (holding ?o) (not (in ?o ?r)))
  )

  (:action PutObject
    :parameters (?o - object ?r - receptacle)
    :precondition (and (holding ?o) (agent-at ?r))
    :effect (and (on ?o ?r) (not (holding ?o)))
  )

  (:action useObject
    :parameters (?o - object)
    :precondition (and (holding ?o) (not (switched-on ?o)))
    :effect (switched-on ?o)
  )

  (:action HeatObject
    :parameters (?o - object ?r - microwaveReceptacle)
    :precondition (and (holding ?o) (agent-at ?r))
    :effect (heated ?o)
  )

  (:action CleanObject
    :parameters (?o - object ?r - receptacle)

  )

  (:action CoolObject
