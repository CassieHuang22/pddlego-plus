  )

    (at-agent ?r - receptacle)
    (at-obj ?o - object ?r - receptacle)
    (holding ?o - object)
    (opened ?r - receptacle)
    (empty-handed)
    (used ?o - object)
    (heated ?o - object)
    (cleaned ?o - object)
    (cooled ?o - object)
    (sliced ?o - object)
  )
  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (and (at-agent ?from))
    :effect (and (not (at-agent ?from)) (at-agent ?to))
  )
  (:action OpenObject
    :parameters (?r - receptacle)
    :precondition (and (at-agent ?r) (not (opened ?r)))
    :effect (and (opened ?r))
  )
  (:action CloseObject
    :parameters (?r - receptacle)
    :precondition (and (at-agent ?r) (opened ?r))
    :effect (and (not (opened ?r)))
  )
