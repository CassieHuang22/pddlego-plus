  (:predicates

    (in ?o - object ?r - receptacle)
    (holding ?a - agent ?o - object)
    (heated ?o - object)
    (clean ?o - object)
    (cooled ?o - object)
    (sliced ?o - object)
  )
  (:action GOTOLOCATION
    :parameters (?a - agent ?from - receptacle ?to - receptacle)
    :precondition (and (at ?a ?from))
    :effect (and (at ?a ?to) (not (at ?a ?from)))
  )
  (:action OPENOBJECT

    :precondition (and (at ?a ?r) (not (open ?r)))
