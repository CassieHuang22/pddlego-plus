    (holding ?o - object)
    (at_agent ?r - receptacle)
  )

  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (and (at_agent ?from))
    :effect (and (not (at_agent ?from)) (at_agent ?to))
  )

  (:action OpenObject
    :parameters (?r - receptacle)
    :precondition (and (closed ?r) (at_agent ?r))

    :parameters (?r - receptacle)
    :precondition (and (open ?r) (at_agent ?r))
